168struct CW_SecureSession {
169 uint8_t aesKey[CW_AESKEY_SIZE];
170 uint8_t macKey[CW_MACKEY_SIZE];
171 uint8_t iv[CW_IV_SIZE];
172
174 CW_SecureSession() {
175 memset(aesKey, 0U, sizeof(aesKey));
176 memset(macKey, 0U, sizeof(macKey));
177 memset(iv, 0U, sizeof(iv));
178 }
179
181 void clear() {
182 CW_Utils::secure_wipe(aesKey, sizeof(aesKey));
183 CW_Utils::secure_wipe(macKey, sizeof(macKey));
184 CW_Utils::secure_wipe(iv, sizeof(iv));
185 }
186};
